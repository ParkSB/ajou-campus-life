\section{Appendix}

\begin{itemize}
  \item $\log_b{a} = c, \quad b^c = a$
  \item $\log_b{1} = 0$
  \item $\log_b{b} = 1$
  \item $\log^0{a} = a$
  \item $\log^1{a} = \log{a}$
  \item $O(n^c \log^k{n}) = O(1), \quad c = 0, \quad k = 0$.
  \item $O(n^c \log^k{n}) = O(n), \quad c = 1, \quad k = 0$.
  \item $O(n^c \log^k{n}) = O(n \log{n}), \quad c = 1, \quad k = 1$.
\end{itemize}
